# Load necessary libraries
library(shiny)
library(tidyverse)

# Define UI
ui <- fluidPage(
  titlePanel("Wine Quality Visualizer"),
  
  sidebarLayout(
    sidebarPanel(
      fileInput("file", "Upload Wine Data CSV File", accept = c(".csv")),
      selectizeInput("wine_id", "Select Wine ID:", choices = NULL),
      actionButton("showDetails", "Show Predicted Quality"),
      hr(),
      radioButtons("plotType", "Select Plot Type:",
                   choices = list("Histogram" = "hist",
                                  "Boxplot" = "box",
                                  "Scatter Plot" = "scatter")),
      conditionalPanel(
        condition = "input.plotType == 'scatter'",
        checkboxInput("regression", "Add Regression Line", value = FALSE)
      ),
      actionButton("plot", "Generate Plot")
    ),
    
    mainPanel(
      textOutput("wineDetails"),
      plotOutput("plotOutput")
    )
  )
)

# Define server logic
server <- function(input, output, session) {
  
  # Reactive expression to read the CSV file
  wine_data <- reactive({
    req(input$file)
    read.csv(input$file$datapath)
  })
  
  # Update the wine_id choices based on the uploaded data
  observe({
    req(wine_data())
    data <- wine_data()
    updateSelectInput(session, "wine_id", choices = unique(data$ID))
  })
  
  # Display details for the selected wine ID
  observeEvent(input$showDetails, {
    req(wine_data())
    data <- wine_data()
    
    selected_data <- data %>%
      filter(ID == input$wine_id)
    
    output$wineDetails <- renderText({
      if (nrow(selected_data) > 0) {
        wine_info <- selected_data[1, ]
        paste("Wine ID: ", wine_info$ID, "\n",
              "Location: ", wine_info$location, "\n",
              "Wine Type: ", wine_info$type, "\n",
              "Acidity: ", wine_info$fixed.acidity, "\n",
              "Sugar: ", wine_info$residual.sugar, "\n",
              "Alcohol: ", wine_info$alcohol, "\n",
              "Predicted Quality: ", wine_info$quality)
      } else {
        "No match found for the selected Wine ID."
      }
    })
  })
  
  # Plotting options based on user input
  plotData <- reactive({
    req(wine_data())
    data <- wine_data() # No filtering by Wine ID here
    data
  })
  
  output$plotOutput <- renderPlot({
    req(input$plot)
    data <- plotData()
    
    if (input$plotType == "hist") {
      ggplot(data, aes(x = quality)) +
        geom_histogram(binwidth = 1, fill = "blue", color = "black") +
        labs(title = "Histogram of Wine Quality", x = "Quality", y = "Count")
      
    } else if (input$plotType == "box") {
      ggplot(data, aes(x = as.factor(quality), y = fixed.acidity)) +
        geom_boxplot(fill = "yellow", color = "black") +
        labs(title = "Boxplot of Acidity by Wine Quality", x = "Quality", y = "Fixed Acidity")
      
    } else if (input$plotType == "scatter") {
      p <- ggplot(data, aes(x = fixed.acidity, y = quality)) +
        geom_point() +
        labs(title = "Scatter Plot of Acidity vs Quality", x = "Fixed Acidity", y = "Quality")
      
      if (input$regression) {
        p <- p + geom_smooth(method = "lm", col = "red")
      }
      
      print(p)
    }
  })
}

# Run the application 
shinyApp(ui = ui, server = server)
